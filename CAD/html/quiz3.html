Plotting Amplitude modulated signal

Tam=linspace(-2*pi,2*pi,1000);
fm=4*10^3;
fc=99.5*10^6;
Ec=40;
Em=100;
Mam=Em/Ec;
Eam=Ec.*(1+Mam*(sin(2*pi*fm.*Tam))).*sin(2*pi*fc.*Tam);
figure
plot(Tam,Eam),xlabel('t[-2\pi 2\pi]'),ylabel('e_{am}'),title('Amplitude Modulated Signal')
